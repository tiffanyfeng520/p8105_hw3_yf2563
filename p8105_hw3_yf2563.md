``` r
accel_data_path = './data/accel_data.csv'

accel_data = 
  read_csv(accel_data_path) %>%
  janitor::clean_names() %>%
  mutate(
    weekday = case_when(
    day == "Saturday" ~ "weekend",
    day == "Sunday"   ~ "weekend",
    TRUE              ~ "weekday"
    )
  ) %>%
  pivot_longer(
    activity_1:activity_1440,
    names_to = "minute",
```
